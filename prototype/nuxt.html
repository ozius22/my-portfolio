📄 services/BaseService.ts

export class BaseService {

  async request<T>(

    url: string, method: string, params: object = {}

  ): Promise<T> {


    // ① Pull base URL from nuxt.config runtimeConfig

    const runtimeConfig = useRuntimeConfig()

    const config: any = {

      baseURL: runtimeConfig.public.apiBaseURL,

      method: method,

      headers: { Accept: 'application/json' },

    }


    // ② GET → query params, else → request body

    if (method.toUpperCase() === 'GET') {

      config.params = params

    } else {

      config.body = params

    }


    try {

      // ③ Nuxt's $fetch handles SSR + client fetch

      return await $fetch<T>(url, config)

    } catch (error: any) {

      // ④ Centralized error handling

      const status = error.response?.status

      switch (status) {

        case 400: case 404: case 422: case 429:

          throw new Error(data?.message || "Validation Error")

        case 500:

          throw new Error("Server error. Contact admin.")

        default:

          throw new Error("Something went wrong.")

      }

    }

  }

}

Error Status Handling in BaseService

HTTP Status Meaning BaseService Response Handle in UI?
200 OK Success Returns typed data T ✓ Use the data
400 Bad Request Invalid input throw new Error(message) Show form error
404 Not Found Resource missing throw new Error(message) Show "not found" UI
422 Unprocessable Validation failed throw new Error(message) Show field errors
429 Too Many Req Rate limited throw new Error(message) Show cooldown notice
500 Server Error Server crashed throw new Error("Server error...") Show global error toast

📄 services/CustomerService.ts

import { BaseService } from './BaseService'


// Define the shape of your data

export interface Customer {

  id: number

  name: string

  email: string

}


export class CustomerService extends BaseService {


  // GET /customers

  getAll() {

    return this.request<Customer[]>(

      '/customers', 'GET'

    )

  }


  // GET /customers/:id

  getById(id: number) {

    return this.request<Customer>(

      `/customers/${id}`, 'GET'

    )

  }


  // POST /customers

  create(data: Partial<Customer>) {

    return this.request<Customer>(

      '/customers', 'POST', data

    )

  }


  // PUT /customers/:id

  update(id: number, data: Partial<Customer>) {

    return this.request<Customer>(

      `/customers/${id}`, 'PUT', data

    )

  }


  // DELETE /customers/:id

  remove(id: number) {

    return this.request<void>(

      `/customers/${id}`, 'DELETE'

    )

  }

}


export default new CustomerService()

// ↑ singleton: import once, use everywhere

📄 pages/customers/index.vue — usage

<script setup lang="ts">

import customerService from '~/services/CustomerService'

import type { Customer } from '~/services/CustomerService'


const customers = ref<Customer[]>([])

const loading = ref(false)

const error = ref<string | null>(null)


const fetchCustomers = async () => {

  loading.value = true

  try {

    customers.value = await customerService.getAll()

  } catch (e: any) {

    error.value = e.message // from BaseService

  } finally {

    loading.value = false

  }

}


onMounted(fetchCustomers)

</script>

📄 nuxt.config.ts — env setup

export default defineNuxtConfig({

  runtimeConfig: {

    // Server-only secrets (not exposed to client)

    apiSecret: '', // ← from env: NUXT_API_SECRET


    public: {

      // Exposed to browser — safe for base URL

      apiBaseURL: '' // ← NUXT_PUBLIC_API_BASE_URL

    }

  }

})

📡 What happens when customerService.create(data) is called:

1 Component calls customerService.create({ name, email })
↓
2 CustomerService.create() calls this.request<Customer>('/customers', 'POST', data)
↓
3 BaseService builds config: method=POST → config.body = data, gets baseURL from runtimeConfig
↓
4 $fetch makes the HTTP call → POST https://api.yourapp.com/customers
↓
5 Success: returns typed Customer object — Error: status switch throws descriptive Error to component
